 \begin{eqnarray*} \cot (a+b) &=&\frac{\cos (a+b)}{\sin (a+b)}=\frac{\cos a\cos b-\sin a\sin b}{ \sin a\cos b+\cos a\sin b} \\ &=&\dfrac{\dfrac{\cos a\cos b-\sin a\sin b}{\sin a\sin b}}{\dfrac{\sin a\cos b+\cos a\sin b}{\sin a\sin b}} \\ &=&\frac{\cot a\cot b-1}{\cot b+\cot a}. \end{eqnarray*}
